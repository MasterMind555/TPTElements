// Element property reference // by boxmein - Powder Toy - Coding tutorial - Contact - About - on GitHub // C++ version - Lua version - Reference // I highly suggest using the Find function of your web browser. // Graphics function - Update function - Element properties

Identifier = "DEFAULT_PT_BCOL"; // Element's name on the new lua API (Not used on the Lua mod) Name = "BCOL"; // Element's in-game name. (4 letters!). Colour = PIXPACK(0x333333); // Element's colour in hexadecimal. (Bits: RRGGBB) // (In lua, skip the PIXPACK and the parentheses.) MenuVisible = 1; // Whether or not it's in a menu. MenuSection = SC_POWDERS; // Which menu it's in. // Options: WALL, ELEC, POWERED, FORCE, EXPLOSIVE, GAS, LIQUID, POWDERS, SOLIDS, NUCLEAR, SPECIAL, LIFE, TOOL, DECO) Enabled = 1; // Whether or not it's even enabled. Keep as 1 unless you have a good reason. Advection = 0.4f; //How much the particle is accelerated by moving air. // Normally 0 for solids, and up to 1 for other elements. // It can be negative, ANAR and DEST do this so it goes towards pressure AirDrag = 0.04f * CFDS; // How much air the particle "drags" along with it. AirLoss = 0.94f; // How much it slows down moving air. 1 for no effect, 0 for full effect. Loss = 0.95f; // How much speed the particle loses every frame. 1 = no loss. 0.5 = half loss. Collision = -0.1f; // When this particle hits something, this value is used to determine new velocity. Gravity = 0.3f; // How fast the particle falls. Negative means it floats. Diffusion = 0.00f; // How much the particle wiggles like GAS. HotAir = 0.000f * CFDS; //How much the particle increases the pressure by. // Another property only for gasses, but VENT/VACU have theirs at (-)0.010f. // An extremely small number, sometimes as small as 0.000001f Falldown = 1; // How does this particle move? (0 = solid. 1 = powder. 2 = liquid) Flammable = 0; // How much this particle burns. (0 = not at all. >0 = burns more.) Explosive = 0; // Does the particle explode? (0 = no. 1 = only with fire. 2 = with pressure > 2.5 or fire.) Meltable = 5; // Does it melt? (0 = no. 1 = yes) Hardness = 2; // How much acid affects the particle. (0 = not at all. >0 = more corrodable) Weight = 90; // Particle weight. Determines which sinks below which. //(1 = gas. 2 = light. 0-49 = liquids. 50-99 = powders. 100 = solid. -1 for energy.) Temperature = R_TEMP+0.0f +273.15f; // How hot does it spawn with? Automatically converted to Celsius. R_TEMP is 22.00° C HeatConduct = 150; // How well it transfers heat. (0 = badly. 255 = very very well) Description = "Broken Coal. Heavy particles. See COAL"; // Element description. Is shown when mouse hovers over element. State = ST_SOLID; // Which state the element is in. (ST_NONE = Special. ST_SOLID = Solid. ST_LIQUID = Liquid. ST_GAS = Gas.) Properties = TYPE_PART; // Element properties. See this for details.

// These next ones use a few special values. // IPL = Impossible Pressure Low - use it to disable element transitions with low pressure. // IPH = Impossible Pressure High - use it to disable element transitions with high pressure. // NT = No Type - use it as a dummy if you don't have a transition with that value. // ITL = Impossible Temperature Low - use it to disable element transitions with low temperature. // ITH = Impossible Temperature High - use it to disable element transitions with high temperature. LowPressure = IPL; // How low should pressure be to trigger a transition? (It can be any number from -255 to 255) LowPressureTransition = NT; // What to transition to when pressure is lower than LowPressure? HighPressure = IPH; // How high should pressure be to trigger a transition? (It can be any number from -255 to 255) HighPressureTransition = NT; // What to transition to when pressure is higher than HighPressure? LowTemperature = ITL; // How low should temp be to trigger a transition? (It can be any number from 0 to 10000) LowTemperatureTransition = NT; // What to transition to when temp is lower than LowTemperature? HighTemperature = ITH; // How high should temp be to trigger a transition? (It can be any number from 0 to 10000) HighTemperatureTransition = NT; // What to transition to when temp is higher than HighTemperature?

Update = &Element_BCOL::update; // The name of the updating function. Change BCOL to what your element's Name is. Graphics = &Element_BCOL::graphics; // The name of the graphics function. Change BCOL to what your element's Name is.

Doesn't support cola. // PMODE_BLOB 0x00000002 --adds a blobby effect, like you were using blob (5) display mode // PMODE_BLUR 0x00000004 --used in liquids in fancy display mode // PMODE_GLOW 0x00000008 --Glow effect, used in elements like DEUT and TRON in fancy display mode // PMODE_SPARK 0x00000010 -- used for things such as GBMB at first, dimmer than other modes // PMODE_FLARE 0x00000020 --BOMB and other similar elements, brighter than PMODE_SPARK // PMODE_LFLARE 0x00000040 --brightest spark mode, used when DEST hits something // PMODE_ADD 0x00000080 --like PMODE_FLAT, but adds color to a pixel, instead of overwriting it. // PMODE_BLEND 0x00000100 --basically the same thing as PMODE_ADD, but has better OpenGL support // PSPEC_STICKMAN 0x00000200 --does nothing, because the stickmen won't get drawn unless it actually is one // NO_DECO 0x00001000 --prevents decoration from showing on the element (used in LCRY) // DECO_FIRE 0x00002000 --Allow decoration to be drawn on using the fire effect (gasses have this set) // FIRE_ADD 0x00010000 --adds a weak fire effect around the element (ex. LAVA/LIGH) // FIRE_BLEND 0x00020000 --adds a stronger fire effect around the element, default for gasses // EFFECT_GRAVIN 0x01000000 --adds a PRTI effect. See below. // EFFECT_GRAVOUT 0x02000000 --adds a PRTO effect. See below. // NOTE: Might take some coding in an update function to get it to work properly, PRTI uses life and ctype to create the effects cola = 200 // Colour transparency/darkness colr = 150 // Red intensity colg = 255-colr // Green intensity colb = 0 // Blue intensity firea = 255 // Alpha on the fire blob firer = 0 // Red intensity of the fire blob fireg = 0 // Green intensity of the fire blob fireb = 255 // Blue intensity of the fire blob
