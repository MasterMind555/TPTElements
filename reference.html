// Element property reference // by boxmein - Powder Toy - Coding tutorial - Contact - About - on GitHub // C++ version - Lua version - Reference Identifier = "DEFAULT_PT_BCOL"; // Element's name on the new lua API (Not used on the Lua mod) Name = "BCOL"; // Element's in-game name. (4 letters!). Colour = PIXPACK(0x333333); // Element's colour in hexadecimal. (Bits: RRGGBB) (In lua, skip the PIXPACK and the parentheses.) MenuVisible = 1; // Whether or not it's in a menu. MenuSection = SC_POWDERS; // Which menu it's in. // Options: WALL, ELEC, POWERED, FORCE, EXPLOSIVE, GAS, LIQUID, POWDERS, SOLIDS, NUCLEAR, SPECIAL, LIFE, TOOL, DECO) Enabled = 1; // Whether or not it's even enabled. Keep as 1 unless you have a good reason. Advection = 0.4f; // How much moving air / wind affects it. AirDrag = 0.04f * CFDS; // How much air the particle "drags" along with it. AirLoss = 0.94f; // How much it slows down moving air. 1 for no effect, 0 for full effect. Loss = 0.95f; // How much speed the particle loses every frame. 1 = no loss. 0.5 = half loss. Collision = -0.1f; // When this particle hits something, this value is used to determine new velocity. Gravity = 0.3f; // How fast the particle falls. Negative means it floats. Diffusion = 0.00f; // How much the particle wiggles like GAS. HotAir = 0.000f * CFDS; // How much the particle increases pressure. Falldown = 1; // How does this particle move? (0 = solid. 1 = powder. 2 = liquid) Flammable = 0; // How much this particle burns. (0 = not at all. >0 = burns more.) Explosive = 0; // Does the particle explode? (0 = no. 1 = only with fire. 2 = with pressure > 2.5 or fire.) Meltable = 5; // Does it melt? (0 = no. 1 = yes) Hardness = 2; // How much acid affects the particle. (0 = not at all. >0 = more corrodable) Weight = 90; // Particle weight. Determines which sinks below which. //(1 = gas. 2 = light. 0-49 = liquids. 50-99 = powders. 100 = solid. -1 for energy.) Temperature = R_TEMP+0.0f +273.15f; // How hot does it spawn with? Automatically converted to Celsius. R_TEMP is 22.00° C HeatConduct = 150; // How well it transfers heat. (0 = badly. 255 = very very well) Description = "Broken Coal. Heavy particles. See COAL"; // Element description. Is shown when mouse hovers over element. State = ST_SOLID; // Which state the element is in. (ST_NONE = Special. ST_SOLID = Solid. ST_LIQUID = Liquid. ST_GAS = Gas.) Properties = TYPE_PART; // Element properties. See this for details. // These next ones use a few special values. // IPL = Impossible Pressure Low - use it to disable element transitions with low pressure. // IPH = Impossible Pressure High - use it to disable element transitions with high pressure. // NT = No Type - use it as a dummy if you don't have a transition with that value. // ITL = Impossible Temperature Low - use it to disable element transitions with low temperature. // ITH = Impossible Temperature High - use it to disable element transitions with high temperature. LowPressure = IPL; // How low should pressure be to trigger a transition? (It can be any number from -255 to 255) LowPressureTransition = NT; // What to transition to when pressure is lower than LowPressure? HighPressure = IPH; // How high should pressure be to trigger a transition? (It can be any number from -255 to 255) HighPressureTransition = NT; // What to transition to when pressure is higher than HighPressure? LowTemperature = ITL; // How low should temp be to trigger a transition? (It can be any number from 0 to 10000) LowTemperatureTransition = NT; // What to transition to when temp is lower than LowTemperature? HighTemperature = ITH; // How high should temp be to trigger a transition? (It can be any number from 0 to 10000) HighTemperatureTransition = NT; // What to transition to when temp is higher than HighTemperature? Update = &Element_BCOL::update; // The name of the updating function. Change BCOL to what your element's Name is. Graphics = &Element_BCOL::graphics; // The name of the graphics function. Change BCOL to what your element's Name is.
